We start with the equations: $$\begin{array}{*{20}{c}} {(1)}\\ {(2)}\\ {(3)} \end{array}\begin{array}{*{20}{c}} {}\\ {}\\ {} \end{array}\begin{array}{*{20}{c}} {\dot T = {k_T}({T_A} - T) - k(\rho )T}\\ {\dot S = {k_S}({S_A} - S) - k(\rho )S}\\ {\rho = - \alpha T + \beta S} \end{array}$$

And after removing dimensions and set parameter values we are left with:

$$\begin{array}{*{20}{c}} {{{\dot T}^*} = 1 - {T^*} - {k_1}(\rho ){T^*}}\\ {{{\dot S}^*} = 0.1(1 - {S^*}) - {k_1}(\rho ){S^*}}\\ {{\rho ^*} = - 0.2{T^*} + {S^*}} \end{array}$$

Where: $${k_1} = \left\{ {\begin{array}{*{20}{c}} {0\,\,\,,\,\,{\rho ^*} \le - 0.01}\\ {5\,\,\,,\,\,{\rho ^*} > - 0.01} \end{array}} \right\}$$

def ode(t, TS):
    T, S = TS

    p = -0.2*T + S
    k1 = 0 if p <= -0.01 else 5

    dTdt = 1 - T - k1*T
    dSdt = 0.1*(1 - S) - k1*S

    return dTdt, dSdt

def kp(T,S):
    p = -0.2*T + S
    k1lo, k1hi = 0, 5
    flip = -0.01
    k1 = 0.5*(k1lo+k1hi + (k1hi-k1lo)*np.tanh((p-flip)/1e-8))
    return k1,p

def Welander(TS, t):
    T, S = TS
    k1,_ = kp(T,S)
    return np.array([1 - T - k1*T,
                    0.1*(1 - S) - k1*S])

def plot_flow_field(ax, f, u_range, v_range, n_grid=100):
    """
    Plots the flow field with line thickness proportional to speed.
    
    Parameters
    ----------
    ax : Matplotlib Axis instance
        Axis on which to make the plot
    f : function for form f(y, t, *args)
        The right-hand-side of the dynamical system.
        Must return a 2-array.
    u_range : array_like, shape (2,)
        Range of values for u-axis.
    v_range : array_like, shape (2,)
        Range of values for v-axis.
    args : tuple, default ()
        Additional arguments to be passed to f
    n_grid : int, default 100
        Number of grid points to use in computing
        derivatives on phase portrait.
        
    Returns
    -------
    output : Matplotlib Axis instance
        Axis with streamplot included.
    """
    
    # Set up u,v space
    u = np.linspace(u_range[0], u_range[1], n_grid)
    v = np.linspace(v_range[0], v_range[1], n_grid)
    uu, vv = np.meshgrid(u, v)

    # Compute derivatives
    u_vel = np.empty_like(uu)
    v_vel = np.empty_like(vv)
    for i in range(uu.shape[0]):
        for j in range(uu.shape[1]):
            u_vel[i,j], v_vel[i,j] = f(np.array([uu[i,j], vv[i,j]]), None)

    # Compute speed
    speed = np.sqrt(u_vel**2 + v_vel**2)

    # Make linewidths proportional to speed,
    # with minimal line width of 0.5 and max of 3
    lw = 0.5 + 2.5 * speed / speed.max()

    # Make stream plot
    ax.streamplot(uu, vv, u_vel, v_vel, linewidth=lw, arrowsize=1.2, 
                  density=1, color='thistle')

    return ax


fig, ax = plt.subplots(1, 1)

ax.set_ylabel('S')
ax.set_aspect('equal')

ax = plot_flow_field(ax, Welander, (0, 1), (0, 1))
